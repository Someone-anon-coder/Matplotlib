# Importing matplotlib.pyplot
# Saving it in memory as plt
import matplotlib.pyplot as plt

# Importing numpy
# Saving it in memory as np
import numpy as np

# Enabling interactive mode
plt.ion()

# Creating data points x from 0 to 10 with 100 points
x = np.linspace(0, 10, 100)

# Looping through different frequencies
for i in range(1, 4):
    
    # Plotting the sine wave with different frequencies
    plt.plot(x, np.sin(i * x))

    # Setting the title of the plot
    plt.title(f'Interactive Plot - Wave {i}')

    # Pausing the plot for 1 second
    plt.pause(1)

    # Saving the plot figure and clearing the plot
    plt.savefig(f'Plots/interactive_plot_{i}.png')
    plt.clf()

# Disabling interactive mode
plt.ioff()
